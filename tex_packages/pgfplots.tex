\usepackage{pgfplots}

    \begin{axis}[
            axis lines = middle,
            xlabel = $x$, ylabel = $y$,
            xmin=-5, xmax=5,
            ymin=-5, ymax=5,
            domain=-5:5, % Domain for the function
            samples=100
        ]
        \addplot[name path=A, blue, thick] {sin(deg(x))+1};
        \addplot[name path=B, domain=-2:2] {0} coordinate (x-axis);
        \addplot[blue!20] fill between[of=A and B];
    \end{axis}
\end{tikzpicture}
